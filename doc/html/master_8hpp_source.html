195  Scalar GoldenSection( T & fcost, Scalar a, Scalar b, Scalar c,
196  Scalar& dt, Scalar& eps ) {
197  static const Scalar resphi = 2.0 - (1.0 + sqrt(5.0))/2.0;
198  static Scalar d;
199  d = ( (c-b)>(b-a) ) ? ( b+resphi*(c-b) ) : ( b-resphi*(b-a) );
200  if ( std::abs(c-a) < eps ) { return (c+a)/2.0; }
201  if ( fcost(d) < fcost(b) ) {
202  if ( (c-b)>(b-a) ) {
203  return GoldenSection( fcost, b, d, c, dt, eps );
204  } else {
205  return GoldenSection( fcost, a, d, b, dt, eps );
206  }
207  } else {
208  if ( (c-b)>(b-a) ) {
209  return GoldenSection( fcost, a, b, d, dt, eps );
210  } else {
211  return GoldenSection( fcost, d, b, c, dt, eps );

229  template < class T , class Scalar >
230  void MinSearch( T & fcost, Scalar t0, Scalar tf, std::vector<Scalar>& lclMin,
231  Scalar dt, Scalar eps ) {
232  int sign1, sign2;
233  Scalar dtt=2.0*dt, maxt=tf-2.0*dt;
234  Scalar c1 = fcost(t0), c2 = fcost(t0+dt), c3 = fcost(t0+dtt);
235 
236  lclMin.push_back( t0 ); /* include endpoint costs */
237  for(Scalar t=t0+dt; t<=maxt; t+=dt) { /* loop over data */
238  sign1 = sgn(c2-c1); // forward diff's to approx deriv
239  sign2 = sgn(c3-c2);
240  if(sign1!=sign2) { // find zero crossings
241  if ( (c1 > c2) || (c3 > c2) ) { // store local min + saddle pts
242  lclMin.push_back( GoldenSection( fcost, t-dt, t, t+dt, dt, eps ) );
243  }
244  }
245  c1 = c2;
246  c2 = c3;
247  c3 = fcost(t+dtt);
248  }
249  lclMin.push_back( tf ); /* include endpoint costs */
250  }
